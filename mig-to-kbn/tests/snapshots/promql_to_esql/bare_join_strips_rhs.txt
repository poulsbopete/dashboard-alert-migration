| WHERE @timestamp >= ?_tstart AND @timestamp <= ?_tend
| WHERE node_hwmon_temp_celsius IS NOT NULL
| STATS node_hwmon_temp_celsius = AVG(node_hwmon_temp_celsius) BY time_bucket = BUCKET(@timestamp, 50, ?_tstart, ?_tend), chip
| SORT time_bucket ASC
