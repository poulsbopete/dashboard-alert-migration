| WHERE @timestamp >= ?_tstart AND @timestamp <= ?_tend
| WHERE http_requests_total IS NOT NULL
| STATS http_requests_total = RATE(http_requests_total, 5m) BY time_bucket = TBUCKET(5 minute)
