source: sum(node_namespace_pod_container:container_cpu_usage_seconds_total:sum_irate{namespace='default'})
feasibility: feasible
---
TS metrics-*
| WHERE @timestamp >= ?_tstart AND @timestamp <= ?_tend
| WHERE k8s.namespace.name == "default"
| WHERE `node_namespace_pod_container:container_cpu_usage_seconds_total:sum_irate` IS NOT NULL
| STATS node_namespace_pod_container_container_cpu_usage_seconds_total_sum_irate = SUM(`node_namespace_pod_container:container_cpu_usage_seconds_total:sum_irate`) BY time_bucket = TBUCKET(5 minute)
| SORT time_bucket ASC
