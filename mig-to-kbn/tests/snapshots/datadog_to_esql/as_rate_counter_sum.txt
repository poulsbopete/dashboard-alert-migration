source: sum:http.requests{*}.as_rate()
status: warning
backend: esql
kibana_type: xy
warning: rate semantics approximated with delta over observed bucket span
---
FROM metrics-*
| WHERE @timestamp >= ?_tstart AND @timestamp <= ?_tend
| STATS value = (MAX(http_requests) - MIN(http_requests)) / (DATE_DIFF("seconds", MIN(@timestamp), MAX(@timestamp)) + 1) BY time_bucket = BUCKET(@timestamp, 50, ?_tstart, ?_tend)
| SORT time_bucket
