source: p99:trace.http.request.duration{*} by {resource_name}
status: ok
backend: esql
kibana_type: xy
---
FROM metrics-*
| WHERE @timestamp >= ?_tstart AND @timestamp <= ?_tend
| STATS value = PERCENTILE(trace_http_request_duration, 99) BY time_bucket = BUCKET(@timestamp, 50, ?_tstart, ?_tend), resource_name
| SORT time_bucket
