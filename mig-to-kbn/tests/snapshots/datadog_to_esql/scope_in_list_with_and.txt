source: avg:system.cpu.user{env:staging AND availability_zone IN (us-east-1a, us-east-1b)} by {availability_zone}
status: ok
backend: esql
kibana_type: xy
---
FROM metrics-*
| WHERE @timestamp >= ?_tstart AND @timestamp <= ?_tend AND deployment.environment == "staging" AND cloud.availability_zone IN ("us-east-1a", "us-east-1b")
| STATS value = AVG(system_cpu_user) BY time_bucket = BUCKET(@timestamp, 50, ?_tstart, ?_tend), cloud.availability_zone
| SORT time_bucket
