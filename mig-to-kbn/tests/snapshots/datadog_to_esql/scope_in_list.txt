source: avg:system.cpu.user{service IN (web, api, worker)} by {host}
status: ok
backend: esql
kibana_type: xy
---
FROM metrics-*
| WHERE @timestamp >= ?_tstart AND @timestamp <= ?_tend AND service.name IN ("web", "api", "worker")
| STATS value = AVG(system_cpu_user) BY time_bucket = BUCKET(@timestamp, 50, ?_tstart, ?_tend), host.name
| SORT time_bucket
