source: top(avg:system.cpu.user{*} by {host}, 50, 'last', 'asc')
status: ok
backend: esql
kibana_type: table
---
FROM metrics-*
| WHERE @timestamp >= ?_tstart AND @timestamp <= ?_tend
| STATS _bucket_value = AVG(system_cpu_user) BY time_bucket = BUCKET(@timestamp, 50, ?_tstart, ?_tend), host.name
| STATS value = LAST(_bucket_value, time_bucket) BY host.name
| SORT value ASC
| LIMIT 50
